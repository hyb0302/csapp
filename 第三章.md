## 3.3

movb   $0xF, (%ebx)

寄存器 ebx 的值是 4 个字节，地址要 8 个字节

movl   %rax, (%rsp)

寄存器 rax 的值是 8 个字节，应该要用 movq

movw   (%rax), 4(%rsp)

mov 源操作数和目标操作数不能同时为内存上的数

movb   %al, %sl

没有 sl 寄存器，只有 sil

movq   %rax, $0x123

不能使用立即数当目标操作数

movl   %eax, %rdx

源操作数的寄存器和目标操作数的寄存器不一样大

movb   %si, 8(%rbp)

寄存器 si 存储的是 2 字节的数据，应该用 movw
